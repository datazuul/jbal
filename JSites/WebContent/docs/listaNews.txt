SELECT p.PID,p.valid,s.CID,c.Type
FROM tblpagine p, tblstrutture s, tblcomponenti c
WHERE p.PID=s.PID
AND s.CID=c.CID
AND p.PID=2
AND c.Type='content';

SELECT co.InsertDate as InsertDate,
	n.startdate as startdate,
	n.enddate as enddate,
	n.CID as CID
FROM tblcontenuti c, tblnews n, tblcomponenti co
WHERE c.PaCID=14
	AND c.StateID=3
	AND co.CID=c.CID
	AND c.CID=n.CID
	AND n.list=1
	AND n.enddate>='20070428'
	AND n.startdate<='20070429'
ORDER BY InsertDate


SELECT s.PID, c.CID, co.InsertDate, n.startdate, n.enddate
FROM tblnews n, 
	tblcontenuti c, 
	tblcomponenti co,
	tblstrutture s
WHERE c.CID=n.CID 
	AND c.StateID=3
	AND co.CID=c.CID
	AND n.list=1
	AND n.enddate>='20070428'
	AND n.startdate<='20070429'
	AND s.CID=c.PaCID
ORDER BY co.InsertDate
